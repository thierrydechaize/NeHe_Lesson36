159 void ReduceToUnit(float vector[3]) // Reduces A Normal Vector (3 Coordinates)
160 { // To A Unit Normal Vector With A Length Of One.
161  float length; // Holds Unit Length
162  // Calculates The Length Of The Vector
163  length = (float)sqrt((vector[0]*vector[0]) + (vector[1]*vector[1]) + (vector[2]*vector[2]));
164 
165  if(length == 0.0f) // Prevents Divide By 0 Error By Providing
166  length = 1.0f; // An Acceptable Value For Vectors To Close To 0.
167 
168  vector[0] /= length; // Dividing Each Element By
169  vector[1] /= length; // The Length Results In A
170  vector[2] /= length; // Unit Normal Vector.
171 }

173 void calcNormal(float v[3][3], float out[3]) // Calculates Normal For A Quad Using 3 Points
174 {
175  float v1[3],v2[3]; // Vector 1 (x,y,z) & Vector 2 (x,y,z)
176  static const int x = 0; // Define X Coord
177  static const int y = 1; // Define Y Coord
178  static const int z = 2; // Define Z Coord
179 
180  // Finds The Vector Between 2 Points By Subtracting
181  // The x,y,z Coordinates From One Point To Another.
182 
183  // Calculate The Vector From Point 1 To Point 0
184  v1[x] = v[0][x] - v[1][x]; // Vector 1.x=Vertex[0].x-Vertex[1].x
185  v1[y] = v[0][y] - v[1][y]; // Vector 1.y=Vertex[0].y-Vertex[1].y
186  v1[z] = v[0][z] - v[1][z]; // Vector 1.z=Vertex[0].y-Vertex[1].z
187  // Calculate The Vector From Point 2 To Point 1
188  v2[x] = v[1][x] - v[2][x]; // Vector 2.x=Vertex[0].x-Vertex[1].x
189  v2[y] = v[1][y] - v[2][y]; // Vector 2.y=Vertex[0].y-Vertex[1].y
190  v2[z] = v[1][z] - v[2][z]; // Vector 2.z=Vertex[0].z-Vertex[1].z
191  // Compute The Cross Product To Give Us A Surface Normal
192  out[x] = v1[y]*v2[z] - v1[z]*v2[y]; // Cross Product For Y - Z
193  out[y] = v1[z]*v2[x] - v1[x]*v2[z]; // Cross Product For X - Z
194  out[z] = v1[x]*v2[y] - v1[y]*v2[x]; // Cross Product For X - Y
195 
196  ReduceToUnit(out); // Normalize The Vectors
197 }

199 void ProcessHelix() // Draws A Helix
200 {
201  GLfloat x; // Helix x Coordinate
202  GLfloat y; // Helix y Coordinate
203  GLfloat z; // Helix z Coordinate
204  GLfloat phi; // Angle
205  GLfloat theta; // Angle
206  GLfloat v,u; // Angles
207  GLfloat r; // Radius Of Twist
208  int twists = 5; // 5 Twists
209  GLfloat glfMaterialColor[4]; // The Material Color
210  GLfloat specular[4]; // Specular Lighting
211 
212  glfMaterialColor[0]=0.4f; // Set The Material Color
213  glfMaterialColor[1]=0.2f;
214  glfMaterialColor[2]=0.8f;
215  glfMaterialColor[3]=1.0f;
216  specular[0]=1.0f; // Sets Up Specular Lighting
217  specular[1]=1.0f;
218  specular[2]=1.0f;
219  specular[3]=1.0f;
220 
221  glLoadIdentity(); // Reset The Modelview Matrix
222  gluLookAt(0, 5, 50, 0, 0, 0, 0, 1, 0); // Eye Position (0,5,50) Center Of Scene (0,0,0), Up On Y Axis
223 
224  glPushMatrix(); // Push The Modelview Matrix
225 
226  glTranslatef(0,0,-50); // Translate 50 Units Into The Screen
227  glRotatef(angle/2.0f,1,0,0); // Rotate By angle/2 On The X-Axis
228  glRotatef(angle/3.0f,0,1,0); // Rotate By angle/3 On The Y-Axis
229 
230  glMaterialfv(GL_FRONT_AND_BACK,GL_AMBIENT_AND_DIFFUSE,glfMaterialColor);
231  glMaterialfv(GL_FRONT_AND_BACK,GL_SPECULAR,specular);
232 
233  r=1.5f; // Radius
234 
235  glBegin(GL_QUADS); // Begin Drawing Quads
236  for(phi=0; phi <= 360; phi+=20.0) // 360 Degrees In Steps Of 20
237  {
238  for(theta=0; theta<=360*twists; theta+=20.0) // 360 Degrees * Number Of Twists In Steps Of 20
239  {
240  v=(phi/180.0f*3.142f); // Calculate Angle Of First Point ( 0 )
241  u=(theta/180.0f*3.142f); // Calculate Angle Of First Point ( 0 )
242 
243  x=(float)(cos(u)*(2.0f+cos(v) ))*r; // Calculate x Position (1st Point)
244  y=(float)(sin(u)*(2.0f+cos(v) ))*r; // Calculate y Position (1st Point)
245  z=(float)((( u-(2.0f*3.142f)) + sin(v) ) * r); // Calculate z Position (1st Point)
246 
247  vertexes[0][0]=x; // Set x Value Of First Vertex
248  vertexes[0][1]=y; // Set y Value Of First Vertex
249  vertexes[0][2]=z; // Set z Value Of First Vertex
250 
251  v=(phi/180.0f*3.142f); // Calculate Angle Of Second Point ( 0 )
252  u=((theta+20)/180.0f*3.142f); // Calculate Angle Of Second Point ( 20 )
253 
254  x=(float)(cos(u)*(2.0f+cos(v) ))*r; // Calculate x Position (2nd Point)
255  y=(float)(sin(u)*(2.0f+cos(v) ))*r; // Calculate y Position (2nd Point)
256  z=(float)((( u-(2.0f*3.142f)) + sin(v) ) * r); // Calculate z Position (2nd Point)
257 
258  vertexes[1][0]=x; // Set x Value Of Second Vertex
259  vertexes[1][1]=y; // Set y Value Of Second Vertex
260  vertexes[1][2]=z; // Set z Value Of Second Vertex
261 
262  v=((phi+20)/180.0f*3.142f); // Calculate Angle Of Third Point ( 20 )
263  u=((theta+20)/180.0f*3.142f); // Calculate Angle Of Third Point ( 20 )
264 
265  x=(float)(cos(u)*(2.0f+cos(v) ))*r; // Calculate x Position (3rd Point)
266  y=(float)(sin(u)*(2.0f+cos(v) ))*r; // Calculate y Position (3rd Point)
267  z=(float)((( u-(2.0f*3.142f)) + sin(v) ) * r); // Calculate z Position (3rd Point)
268 
269  vertexes[2][0]=x; // Set x Value Of Third Vertex
270  vertexes[2][1]=y; // Set y Value Of Third Vertex
271  vertexes[2][2]=z; // Set z Value Of Third Vertex
272 
273  v=((phi+20)/180.0f*3.142f); // Calculate Angle Of Fourth Point ( 20 )
274  u=((theta)/180.0f*3.142f); // Calculate Angle Of Fourth Point ( 0 )
275 
276  x=(float)(cos(u)*(2.0f+cos(v) ))*r; // Calculate x Position (4th Point)
277  y=(float)(sin(u)*(2.0f+cos(v) ))*r; // Calculate y Position (4th Point)
278  z=(float)((( u-(2.0f*3.142f)) + sin(v) ) * r); // Calculate z Position (4th Point)
279 
280  vertexes[3][0]=x; // Set x Value Of Fourth Vertex
281  vertexes[3][1]=y; // Set y Value Of Fourth Vertex
282  vertexes[3][2]=z; // Set z Value Of Fourth Vertex
283 
284  calcNormal(vertexes,normal); // Calculate The Quad Normal
285 
286  glNormal3f(normal[0],normal[1],normal[2]); // Set The Normal
287 
288  // Render The Quad
289  glVertex3f(vertexes[0][0],vertexes[0][1],vertexes[0][2]);
290  glVertex3f(vertexes[1][0],vertexes[1][1],vertexes[1][2]);
291  glVertex3f(vertexes[2][0],vertexes[2][1],vertexes[2][2]);
292  glVertex3f(vertexes[3][0],vertexes[3][1],vertexes[3][2]);
293  }
294  }
295  glEnd(); // Done Rendering Quads
296 
297  glPopMatrix(); // Pop The Matrix
298 }

339 void DrawBlur(int times, float inc) // Draw The Blurred Image
340 {
341  float spost = 0.0f; // Starting Texture Coordinate Offset
342  float alphainc = 0.3f / times; // Fade Speed For Alpha Blending
343  float alpha = 0.1f; // Starting Alpha Value
344  int num = 0;
345 
346  // Disable AutoTexture Coordinates
347  glDisable(GL_TEXTURE_GEN_S);
348  glDisable(GL_TEXTURE_GEN_T);
349 
350  glEnable(GL_TEXTURE_2D); // Enable 2D Texture Mapping
351  glDisable(GL_DEPTH_TEST); // Disable Depth Testing
352  glBlendFunc(GL_SRC_ALPHA,GL_ONE); // Set Blending Mode
353  //glBlendFunc(GL_ONE,GL_ONE); // Set Blending Mode
354  glEnable(GL_BLEND); // Enable Blending
355  glBindTexture(GL_TEXTURE_2D,BlurTexture); // Bind To The Blur Texture
356  ViewOrtho(); // Switch To An Ortho View
357 
358  alphainc = alpha / times; // alphainc=0.2f / Times To Render Blur
359 
360  glBegin(GL_QUADS); // Begin Drawing Quads
361  for (num = 0;num < times;num++) // Number Of Times To Render Blur
362  {
363  glColor4f(1.0f, 1.0f, 1.0f, alpha); // Set The Alpha Value (Starts At 0.2)
364  glTexCoord2f(0+spost,1-spost); // Texture Coordinate ( 0, 1 )
365  glVertex2f(0,0); // First Vertex ( 0, 0 )
366 
367  glTexCoord2f(0+spost,0+spost); // Texture Coordinate ( 0, 0 )
368  glVertex2f(0,480); // Second Vertex ( 0, 480 )
369 
370  glTexCoord2f(1-spost,0+spost); // Texture Coordinate ( 1, 0 )
371  glVertex2f(640,480); // Third Vertex ( 640, 480 )
372 
373  glTexCoord2f(1-spost,1-spost); // Texture Coordinate ( 1, 1 )
374  glVertex2f(640,0); // Fourth Vertex ( 640, 0 )
375 
376  spost += inc; // Gradually Increase spost (Zooming Closer To Texture Center)
377  alpha = alpha - alphainc; // Gradually Decrease alpha (Gradually Fading Image Out)
378  }
379  glEnd(); // Done Drawing Quads
380 
381  ViewPerspective(); // Switch To A Perspective View
382 
383  glEnable(GL_DEPTH_TEST); // Enable Depth Testing
384  glDisable(GL_TEXTURE_2D); // Disable 2D Texture Mapping
385  glDisable(GL_BLEND); // Disable Blending
386  glBindTexture(GL_TEXTURE_2D,0); // Unbind The Blur Texture
387 }
